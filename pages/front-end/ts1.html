# js中的七个基本类型

在JavaScript中我们就见过七个基本数据类型，它们是：布尔类型、数值类型、字符串、null、undefined、object以及ES6中新增的symbol。
但在TypeScript中它们都有对应的类型关键字，对应关系和语法为:

# *布尔类型：boolean

布尔类型的变量值只能是true或者false

let bool: boolean = true
let bool: boolean = 123  //error 类型错误不能将123赋给bool

# *数值类型：number

所有数字都是浮点数，而没有int或者float类型，所以只有一个number类型。

let num: number;  
num = 123;  
num = "123"; // error 不能将类型"123"分配给类型"number"  
num = 0b1111011; //  二进制的123  
num = 0o173; // 八进制的123  
num = 0x7b; // 十六进制的123`

# *字符串类型：string

支持模板字符串

let firstName: string = 'david'
let secondName: string = 'shen'
let str: string = `${firstName} ${secondName}`
console.log(str)  // 'david shen'
