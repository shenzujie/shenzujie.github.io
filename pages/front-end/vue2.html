当vue接触的多了之后，你可能也会到自己封装组件的程度，试想每个页面的功能级模块全部拆分成组件，然后后续请求后台数据传入进去或者自己模拟数据，是多么方便的一件事情。 每当我们需要修改的时候，只需维护那一个功能性组件即可，不需要这个功能了，只需要从页面中删除这个组件的引用即可.
下面的例子是在vue中调用全局弹窗

# 在main.js中注册

import Vue from 'vue'
import App from './App.vue'
import LoginDialog from '@/components/loginDialog/index.vue'

const loginDialog = new Vue(LoginDialog)
loginDialog.$mount(document.createElement('div'))
document.body.appendChild(loginDialog.$el)
Vue.prototype.$loginDialog = loginDialog
export default loginDialog

new Vue({
  render: h => h(App)
}).$mount('#app')

# 在组件中引用

在vue中可以直接通过this.$ref加方法就可以调用组件方法来调用组件

<template>
    <div class=''>
       <el-button @click="dialogBtn">弹框</el-button>
    </div>
</template>

<script>
export default {
    components: {},
    data() {
        return {

        };
    },
    methods: {
        dialogBtn(){
            this.$loginDialog.open()
        }
    },
}
</script>

# 子组件代码

<template>
    <div class=''>
        <el-dialog
        title="提示"
        :visible.sync="dialogVisible"
        width="30%"
        :before-close="handleClose">
        <span>这是一段信息</span>
        <span slot="footer" class="dialog-footer">
            <el-button @click="dialogVisible = false">取 消</el-button>
            <el-button type="primary" @click="dialogVisible = false">确 定</el-button>
        </span>
        </el-dialog>
    </div>
</template>

<script>
//import 《组件名称》 from '《组件路径》';

export default {
    components: {},
    data() {
        return {
            dialogVisible: false
        };
    },
    methods: {
        open(){
            this.dialogVisible = true
        },
        handleClose(done) {
            this.$confirm('确认关闭？')
            .then(_ => {
                done();
            })
            .catch(_ => {});
        }
    },
}
</script>
